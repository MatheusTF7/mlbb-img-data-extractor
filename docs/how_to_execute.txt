# Imagem Específica

python main.py -i ./images/win.jpeg -p "MTF7" --tesseract-cmd "C:\Program Files\Tesseract-OCR\tesseract.exe"

python main.py -i ./images/win.jpeg --all-players --tesseract-cmd "C:\Program Files\Tesseract-OCR\tesseract.exe"

python main.py --generate-config

## Listar perfis de resolução
python main.py --list-profiles
